# Примеры команд для тестирования программы
# Каждая команда будет циклически выполняться с интервалом в 10 секунд

# Windows
dir            # вывод содержимого текущего каталога
echo "Hello"   # вывод текста
whoami         # вывод имени текущего пользователя
ver            # вывод версии операционной системы
time /t        # вывод текущего времени
date /t        # вывод текущей даты
ipconfig       # вывод сетевой конфигурации
systeminfo     # вывод информации о системе (займет больше времени)
